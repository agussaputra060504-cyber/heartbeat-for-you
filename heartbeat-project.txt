    <script>
        const startBtn = document.getElementById('start-btn');
        const welcomeScreen = document.getElementById('welcome-screen');
        const canvas = document.getElementById('heart');
        const messageCard = document.getElementById('message-card');
        const closeBtn = document.getElementById('close-btn');
        const ctx = canvas.getContext('2d');

        let animationFrameId;
        const words = ["love you", "forever", "always", "heartbeat", "my love", "kamu", "je t'aime", "te amo", "mine"];

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);

        function getHeartPoint(t, scale) {
            const x = 16 * Math.pow(Math.sin(t), 3);
            const y = -(13 * Math.cos(t) - 5 * Math.cos(2 * t) - 2 * Math.cos(3 * t) - Math.cos(4 * t));
            return { x: x * scale, y: y * scale };
        }

        class Particle {
            constructor(canvasWidth, canvasHeight) {
                this.reset(canvasWidth, canvasHeight);
            }

            reset(canvasWidth, canvasHeight) {
                this.t = Math.random() * Math.PI * 2;
                this.word = words[Math.floor(Math.random() * words.length)];
                this.scale = Math.random() * 8 + 4;
                
                const point = getHeartPoint(this.t, this.scale);
                this.targetX = canvasWidth / 2 + point.x;
                this.targetY = canvasHeight / 2 + point.y;
                
                this.x = Math.random() * canvasWidth;
                this.y = Math.random() * canvasHeight;
                
                this.speed = Math.random() * 0.02 + 0.01;
                this.opacity = Math.random() * 0.5 + 0.5;
                this.size = Math.random() * 10 + 8;
            }

            update(beatFactor) {
                const point = getHeartPoint(this.t, this.scale * beatFactor);
                const actualTargetX = canvas.width / 2 + point.x;
                const actualTargetY = canvas.height / 2 + point.y;

                this.x += (actualTargetX - this.x) * this.speed;
                this.y += (actualTargetY - this.y) * this.speed;
            }

            draw() {
                ctx.fillStyle = `rgba(255, 42, 116, ${this.opacity})`;
                ctx.font = `${this.size}px 'Courier New'`;
                ctx.fillText(this.word, this.x, this.y);
            }
        }

        const particles = [];
        function initParticles() {
            particles.length = 0;
            const particleCount = 120;
            for (let i = 0; i < particleCount; i++) {
                particles.push(new Particle(canvas.width, canvas.height));
            }
        }

        let tick = 0;
        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            tick += 0.05;
            const beatFactor = 1 + Math.abs(Math.sin(tick)) * 0.15;

            particles.forEach(p => {
                p.update(beatFactor);
                p.draw();
            });

            animationFrameId = requestAnimationFrame(animate);
        }

        startBtn.addEventListener('click', () => {
            welcomeScreen.classList.add('hidden');
            canvas.style.display = 'block';
            resizeCanvas();
            initParticles();
            animate();

            setTimeout(() => {
                messageCard.classList.remove('hidden');
                setTimeout(() => {
                    messageCard.classList.add('show');
                }, 100);
            }, 6000);
        });

        closeBtn.addEventListener('click', () => {
            messageCard.classList.remove('show');
            setTimeout(() => {
                messageCard.classList.add('hidden');
            }, 1000);
        });
    </script>
